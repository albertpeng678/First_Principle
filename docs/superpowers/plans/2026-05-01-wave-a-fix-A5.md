- [ ] **Step 2: PNG handler**

在 final report binding 區（找 `circles-final-home` `addEventListener` 同處）加：

```js
document.getElementById('btn-export-png')?.addEventListener('click', async function() {
  var btn = this;
  var orig = btn.innerHTML;
  btn.disabled = true;
  btn.innerHTML = '<span class="dot"></span> 產生中…';
  try {
    var mod = await import('https://esm.sh/html2canvas@1.4.1');
    var html2canvas = mod.default || mod;
    var target = document.querySelector('.circles-final-report') || document.querySelector('[data-view="circles"]');
    if (!target) throw new Error('report container not found');
    var canvas = await html2canvas(target, { backgroundColor: '#ffffff', scale: 2 });
    var dataUrl = canvas.toDataURL('image/png');
    var a = document.createElement('a');
    a.href = dataUrl;
    a.download = 'pm-drill-report-' + Date.now() + '.png';
    document.body.appendChild(a); a.click(); a.remove();
  } catch (e) {
    console.warn('[png-export] failed:', e && e.message);
    showToast && showToast('PNG 匯出失敗，請改用截圖');
  } finally {
    btn.disabled = false;
    btn.innerHTML = orig;
  }
});
```

如果 `showToast` 不存在，inline `alert('PNG 匯出失敗，請改用截圖')` 為最低 fallback（不 blocking）。
