```js
function localDraftKey(sessionId) {
  return 'circles_draft_' + (sessionId || 'pending') + '_v1';
}

function saveCirclesDraftToLocal() {
  try {
    var key = localDraftKey(AppState.circlesSession && AppState.circlesSession.id);
    var payload = {
      stepDrafts: AppState.circlesStepDrafts || {},
      frameworkDraft: AppState.circlesFrameworkDraft || {},
      savedAt: Date.now(),
    };
    localStorage.setItem(key, JSON.stringify(payload));
  } catch (e) {
    console.warn('[circles local-draft] write failed:', e && e.message);
  }
}

function loadCirclesDraftFromLocal(sessionId) {
  try {
    var raw = localStorage.getItem(localDraftKey(sessionId));
    if (!raw) return null;
    return JSON.parse(raw);
  } catch (e) { return null; }
}

function clearCirclesDraftLocal(sessionId) {
  try { localStorage.removeItem(localDraftKey(sessionId)); } catch (e) {}
}
```
