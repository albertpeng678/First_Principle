```js
// @ts-check
const { test, expect } = require('@playwright/test');
const BASE_URL = process.env.PMDRILL_BASE_URL || 'http://localhost:4000';

test.describe('D-1 · simulation 卡 copy', () => {
  test('home simulation 卡描述含「提示與範例隨時可看」', async ({ page }) => {
    await page.goto(BASE_URL + '/?onboarding=0');
    const desc = page.locator('.circles-mode-card-desc');
    const count = await desc.count();
    expect(count).toBeGreaterThan(0);
    let found = false;
    for (let i = 0; i < count; i++) {
      const t = await desc.nth(i).innerText();
      if (t.includes('提示與範例隨時可看')) found = true;
      expect(t).not.toContain('無提示');
    }
    expect(found).toBe(true);
  });
});

test.describe('D-2 · 錯誤訊息 alert 卡', () => {
  test.use({ viewport: { width: 360, height: 720 } });
  test('Mobile-360 送出空 form 顯示 alert 卡 (不在 submit-bar 內)', async ({ page }) => {
    await page.goto(BASE_URL + '/?onboarding=0');
    await page.locator('.circles-q-card').first().click();
    await page.locator('.circles-q-confirm-btn').first().click();
    await page.waitForSelector('#circles-p1-submit', { timeout: 5000 });
    await page.click('#circles-p1-submit'); // 空 form 送出
    const err = page.locator('#circles-p1-preflight-err');
    await expect(err).toBeVisible({ timeout: 3000 });
    // alert 卡 NOT 在 .circles-submit-bar 內
    const insideBar = await err.evaluate(el => !!el.closest('.circles-submit-bar'));
    expect(insideBar).toBe(false);
    // 內容含「還有 N 個欄位」
    const txt = await err.innerText();
    expect(txt).toMatch(/還有.+個欄位需要填寫/);
  });
});

test.describe('D-5 · system-ui 字型', () => {
  test('--c-font-sans token 為 system-ui chain', async ({ page }) => {
    await page.goto(BASE_URL + '/?onboarding=0');
    const v = await page.evaluate(() =>
      getComputedStyle(document.documentElement).getPropertyValue('--c-font-sans').trim());
    expect(v.toLowerCase()).toContain('system-ui');
    expect(v.toLowerCase()).toContain('pingfang');
  });

  test('body computed font-family 不含 DM Sans 為 first', async ({ page }) => {
    await page.goto(BASE_URL + '/?onboarding=0');
    const ff = await page.evaluate(() => getComputedStyle(document.body).fontFamily);
    expect(ff.toLowerCase().startsWith('"dm sans"')).toBe(false);
    expect(ff.toLowerCase()).toContain('system-ui');
  });

  test('Phase-4 grade letter 仍 Instrument Serif', async ({ page, request }) => {
    // 直接讀 style.css 找 .grade-letter rule 含 Instrument Serif
    const css = await (await request.get(BASE_URL + '/style.css')).text();
    expect(css).toMatch(/\.grade-letter[^}]*Instrument\s*Serif/);
  });
});
```

```js
      var emptyLabels = emptyFields.map(function(el) {

      var msgHtml = uniqLabels.length
```
