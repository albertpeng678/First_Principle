```js
// @ts-check
const { test, expect } = require('@playwright/test');
const BASE_URL = process.env.PMDRILL_BASE_URL || 'http://localhost:4000';

test.describe('MASTER-025 登入成功跳首頁', () => {
  test('?view=login 進來，模擬 SIGNED_IN 後 view 切回 circles', async ({ page }) => {
    await page.goto(BASE_URL + '/?view=login&onboarding=0');
    await expect(page.locator('[data-view="login"]')).toBeVisible({ timeout: 5000 });

    // 模擬 onAuthStateChange 內部呼叫
    await page.evaluate(() => {
      // 用既有 supabase callback 直接觸發 SIGNED_IN
      // 若 supabase 不可用，直接設 AppState 模擬效果
      const fakeSession = { access_token: 'test-token', user: { id: 'test-user' } };
      const cb = window._authCallback;
      if (cb) cb('SIGNED_IN', fakeSession);
      else {
        // fallback: 設 state + 模擬 handler 行為
        window.AppState.mode = 'auth';
        window.AppState.accessToken = fakeSession.access_token;
        window.AppState.user = fakeSession.user;
        if (window.AppState.view === 'login' || window.AppState.view === 'register') {
          window.AppState.view = 'circles';
        }
        window.render && window.render();
      }
    });

    await page.waitForTimeout(800);
    const view = await page.evaluate(() => window.AppState.view);
    expect(view).toBe('circles');
    await expect(page.locator('[data-view="circles"]')).toBeVisible({ timeout: 5000 });
  });

  test('?view=circles 進來，SIGNED_IN 不影響 view', async ({ page }) => {
    await page.goto(BASE_URL + '/?onboarding=0');
    await expect(page.locator('[data-view="circles"]')).toBeVisible();
    await page.evaluate(() => {
      window.AppState.mode = 'auth';
      window.AppState.accessToken = 'test-token';
      // 模擬 handler：不在 login/register 時不切 view
      if (window.AppState.view === 'login' || window.AppState.view === 'register') {
        window.AppState.view = 'circles';
      }
      window.render && window.render();
    });
    const view = await page.evaluate(() => window.AppState.view);
    expect(view).toBe('circles');
  });
});
```

```js
describe('A5 migrateGuestSessions — 23505 conflict path', () => {
  let mockFetch;
  beforeEach(() => {
    mockFetch = jest.fn();
    global.fetch = mockFetch;
  });

  test('23505 conflict 增加 conflicts 計數，不噴 error', async () => {
    mockFetch.mockResolvedValue({
      ok: true,
      json: async () => ({ ok: true, claimed: 0, conflicts: 1, copied: 0 }),
    });

    const data = await result.json();
    expect(data.conflicts).toBe(1);
    expect(data.ok).toBe(true);
  });

  test('成功 migration 三 bucket 都搬', async () => {
    mockFetch.mockResolvedValue({
      ok: true,
      json: async () => ({ ok: true, claimed: 5, conflicts: 0, copied: 2, buckets: ['circles', 'nsm', 'practice'] }),
    });
    const result = await fetch('/api/migrate-guest', { method: 'POST' });
    const data = await result.json();
    expect(data.claimed).toBeGreaterThan(0);
    expect(data.buckets.length).toBe(3);
```
